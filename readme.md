```R
kwnn <- function(xl, k, q) {	  
  #	–≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∫–ª–∞—Å—Å –æ–±—ä–µ–∫—Ç–∞ —á–∞—â–µ –≤—Å–µ–≥–æ –≤—Å—Ç—Ä–µ—á–∞—é—â–µ–π—Å—è
  
  n <- ncol(xl)
  classes <- xl[1:k, n] 
  table <- table(classes)
  table[1:length(table) ] <- 0
  for(i in names(table))
    for(j in 1:k) # –ø–æ j-—Ç—ã–º —Å–æ—Å–µ–¥—è–º
      if(i == xl[j, n]) # i - –∫–ª–∞—Å—Å—ã
        table[i] =  table[i] + q^j # –¥–æ–±–∞–≤–ª—è–µ–º –≤–µ—Å
  class <- names(which.max(table)) 
  return (class)	  
}
```

```R
loo <- function(xl) {
  #	—Ñ—É–Ω–∫—Ü–∏—è –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∞—Å—Å–∏–≤ —Å—Ä–µ–¥–Ω–∏—Ö –æ—à–∏–±–æ–∫
  l <- nrow(xl)
  n <- ncol(xl)
  value_q <- seq(0.1, 1, 0.1) # –ø–µ—Ä–µ–±–∏—Ä–∞–µ–º q 
  Sum <- matrix(0, l-1, length(value_q)) # –º–∞—Ç—Ä–∏—Ü–∞, –≥–¥–µ k —ç—Ç–æ —Å—Ç—Ä–æ–∫–∏ –∞ q - —Å—Ç–æ–ª–±—Ü—ã
  for (i in 1:l){
    z <- xl[i, 1 : (n-1)]
    xl1 <- sort_ojects_by_dist(xl[-i, ], z)		
    for(k in 1:(l-1)){
      int_q <- 1
      for(q in value_q){
       class <- kwnn(xl1, k, q)
       if(xl[i, n] != class) 
        Sum[k, int_q] <- Sum[k, int_q] + 1/l 	
       int_q <- int_q + 1 
      }
    }
  }
  return(Sum)
```

```R
      if(xl[i, n] != class || class == 0) 
        Sum[cnt] <- Sum[cnt] + 1/l 	
      cnt <- cnt +1  
    }
  }
  return(Sum)
}
```

```R
potencial <- function(xl, h, error, ker_function){
```
